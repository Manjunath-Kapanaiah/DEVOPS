	> sudo nano /etc/default/jenkins

	JAVA_ARGS="-Xms512m -Xmx2048m"

	sudo systemctl restart jenkins

Force Disk Cleanup if /var/lib/jenkins > 10GB::

Check Disk Usage: 	df -h

Check usage of:		/var/lib/jenkins

Create Cleanup Script:

		> sudo vim cleanup.sh

#!/bin/bash

USAGE=$(du -s /var/lib/jenkins | cut -f1)
LIMIT=$((10 * 1024 * 1024))  # 10GB in KB

if [ "$USAGE" -gt "$LIMIT" ]; then
    echo "Jenkins exceeds 10GB. Cleaning old builds..."

    find /var/lib/jenkins/jobs/ -type d -name builds -exec rm -rf {}/* \;

    echo "Cleanup completed."
else
    echo "Disk usage under control."
fi



chmod +x cleanup.sh
./cleanup.sh
